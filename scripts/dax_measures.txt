Percentual Language Desconhecido = 
VAR total = COUNTROWS('repos')
VAR semLang = COUNTROWS(FILTER('repos', 'repos'[language] = "Desconhecido" || ISBLANK('repos'[language])))
RETURN DIVIDE(semLang, total, 0)
